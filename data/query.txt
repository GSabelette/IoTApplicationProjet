PREFIX ns: <http://www.semanticweb.org/valentin/ontologies/2022/0/untitled-ontology-7#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>

SELECT ?concentration
WHERE {
 ?mesure rdf:type ns:mesure.
 ?mesure ns:concentration ?concentration.
 ?mesure ns:location ?location.
 FILTER regex(?location, %PARAMETER%).
 }
